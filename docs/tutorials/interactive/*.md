# Interactive Documentation & Live Playground

## Overview

This tutorial transforms forge's static documentation into an interactive experience with live code execution, model comparison tools, and a built-in playground for testing fine-tuning configurations without local setup. Powered by JupyterLite and Pyodide, you can run Python code directly in your browser.

## Getting Started

### Prerequisites

Before using the interactive features, ensure you have:
- Modern web browser (Chrome, Firefox, Safari, Edge)
- JavaScript enabled
- Stable internet connection for initial Pyodide loading

### Basic Fine-tuning Example

Experience forge's capabilities directly in your browser. This example demonstrates a minimal fine-tuning setup:

```{pyodide}
import json
from typing import Dict, Any

class Mockforge:
    """Simplified forge for demonstration"""
    
    def __init__(self, model_name: str = "llama-3-8b"):
        self.model_name = model_name
        self.config = {
            "model": model_name,
            "dataset": "alpaca",
            "epochs": 3,
            "batch_size": 4,
            "learning_rate": 2e-5,
            "lora_rank": 8
        }
    
    def update_config(self, **kwargs):
        """Update configuration parameters"""
        self.config.update(kwargs)
        return self.config
    
    def simulate_training(self) -> Dict[str, Any]:
        """Simulate training process"""
        import random
        
        # Simulate training metrics
        epochs = self.config["epochs"]
        metrics = {
            "train_loss": [],
            "eval_loss": [],
            "learning_rate": []
        }
        
        base_loss = 2.5
        for epoch in range(epochs):
            # Simulate decreasing loss
            train_loss = base_loss * (0.85 ** epoch) + random.uniform(-0.1, 0.1)
            eval_loss = train_loss * 1.1 + random.uniform(-0.05, 0.05)
            lr = self.config["learning_rate"] * (0.9 ** epoch)
            
            metrics["train_loss.append(round(train_loss, 4))
            metrics["eval_loss"].append(round(eval_loss, 4))
            metrics["learning_rate"].append(lr)
        
        return {
            "status": "completed",
            "model": self.model_name,
            "config": self.config,
            "metrics": metrics,
            "final_train_loss": metrics["train_loss"][-1],
            "final_eval_loss": metrics["eval_loss"][-1]
        }

# Initialize and run example
factory = Mockforge("llama-3-8b")
print("Initial configuration:")
print(json.dumps(factory.config, indent=2))

# Update configuration
factory.update_config(
    epochs=5,
    learning_rate=1e-5,
    lora_rank=16
)

print("\nUpdated configuration:")
print(json.dumps(factory.config, indent=2))

# Run training simulation
results = factory.simulate_training()
print("\nTraining Results:")
print(f"Status: {results['status']}")
print(f"Final Training Loss: {results['final_train_loss']}")
print(f"Final Evaluation Loss: {results['final_eval_loss']}")
```

import pandas as pd
from dataclasses import dataclass

    training_time: str
    quality_score: float

def compare_models(configs: List[ModelConfig]):
    """Compare different model configurations"""
    
    # Create comparison table
    data = []
    for config in configs:
        data.append({
            "Model": config.name,
            "Parameters": config.parameters,
            "Context Length": config.context_length,
            "VRAM Required": config.vram_required,
            "Training Time": config.training_time,
            "Quality Score": f"{config.quality_score:.1f}/10"
        })
    
    df = pd.DataFrame(data)
    
    # Display table
    print("Model Comparison Table:")
    print("=" * 80)
    print(df.to_string(index=False))
    print("=" * 80)
    
    # Create visualization
    fig, axes = plt.subplots(1, 2, figsize=(12, 5))
    
    # Quality vs Parameters
    param_values = [float(c.parameters.split()[0]) for c in configs]
    quality_scores = [c.quality_score for c in configs]
    
    axes[0].bar([c.name for c in configs], quality_scores, color='skyblue')
    axes[0].set_xlabel('Model')
    axes[0].set_ylabel('Quality Score')
    axes[0].set_title('Model Quality Comparison')
    axes[0].set_ylim(0, 10)
    axes[0].tick_params(axis='x', rotation=45)
    
    # Training Time vs Parameters
    training_times = []
    for c in configs:
        if "hours" in c.training_time:
            training_times.append(float(c.training_time.split()[0]))

            training_times.append(float(c.training_time.split()[0]) / 60)

    axes[1].set_title('Training Time vs Model Size')

    plt.tight_layout()
    plt.show()
    
    return df

            "llama-3-70b": {"base_vram": 80, "per_sample": 2.0},

            model = "llama-3-8b"  # Default

        lr_factor = 1.0 if 1e-5 <= config["training"]["learning_rate"] <= 5e-5 else 0.9
        lora_factor = 1.0 + (config["lora"]["rank"] - 8) * 0.01

            "train_loss": [],
            "eval_loss": [],
            "accuracy": []

            # Simulate improvement
            improvement = (epoch + 1) / epochs
            train_loss = base["loss"] * (1 - improvement * 0.3) * lr_factor * lora_factor
            eval_loss = train_loss * (1.1 + random.uniform(-0.05, 0.05))
            accuracy = base["accuracy"] * (1 + improvement * 0.2) * lr_factor * lora_factor * epoch_factor
            
            metrics["train_loss"].append(round(train_loss, 4))
            metrics["eval_loss"].append(round(eval_loss, 4))
            metrics["accuracy"].append(round(accuracy, 4))

import sys
import os

# Simulate importing from existing modules
def calculate_learning_rate(base_lr: float, batch_size: int, 
                          world_size: int = 1, strategy: str = "linear") -> float:
    """
    Calculate learning rate based on batch size and strategy.
    Based on scripts/stat_utils/cal_lr.py
    """
    effective_batch_size = batch_size * world_size
    
    if strategy == "linear":
        # Linear scaling rule
        return base_lr * (effective_batch_size / 32)
    elif strategy == "sqrt":
        # Square root scaling rule
        return base_lr * (effective_batch_size / 32) ** 0.5
    elif strategy == "constant":
        return base_lr
    else:
        raise ValueError(f"Unknown strategy: {strategy}")

# Example usage
base_lr = 2e-5
batch_size = 4
world_size = 1

print("Learning Rate Calculator")
print("="*50)

print(f"Batch Size: {batch_size}")
print(f"World Size: {world_size}")
print()

strategies = ["linear", "sqrt", "constant"]
for strategy in strategies:
    lr = calculate_learning_rate(base_lr, batch_size, world_size, strategy)
    print(f"{strategy.capitalize()} scaling: {lr:.2e}")
